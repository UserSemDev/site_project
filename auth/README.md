Для работы с проектом необходим файл .env

```text
    # Settings MongoDB
    MONGO_INITDB_ROOT_USERNAME=
    MONGO_INITDB_ROOT_PASSWORD=
    MONGO_INITDB_DATABASE=
    MONGO_HOST=
    MONGO_PORT=
    
    # Settings Djnago
    SECRET_KEY=
    DEBUG=
    DJANGO_ALLOWED_HOSTS=
    
    # Settings JWT
    SECRET_KEY_JWT=
    JWT_ALGORITHM=
```
